package org.example.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.reference.CtTypeReference;
import spoon.reflect.factory.Factory;
import spoon.support.sniper.SniperJavaPrettyPrinter;

/**
 * Refactoring rule for SpongeAPI v7 to v8 migration based on dependency changes.
 * Handles the removal of CommandSource and TextSerializers.
 */
public class SpongeRefactoring {

    public static class SpongeTypeReplacementProcessor extends AbstractProcessor<CtTypeReference<?>> {

        @Override
        public boolean isToBeProcessed(CtTypeReference<?> candidate) {
            // Defensive check: ignore nulls or primitives (optimization)
            if (candidate == null || candidate.isPrimitive()) {
                return false;
            }

            // Get Qualified Name safely
            String qName = candidate.getQualifiedName();

            // Match 1: CommandSource
            if (qName.equals("org.spongepowered.api.command.CommandSource")) {
                return true;
            }

            // Match 2: TextSerializers
            if (qName.equals("org.spongepowered.api.text.serializer.TextSerializers")) {
                return true;
            }

            return false;
        }

        @Override
        public void process(CtTypeReference<?> candidate) {
            Factory factory = getFactory();
            String qName = candidate.getQualifiedName();
            CtTypeReference<?> newTypeRef = null;

            if (qName.equals("org.spongepowered.api.command.CommandSource")) {
                // Logic: CommandSource in v7 is largely replaced by Adventure's Audience in v8
                newTypeRef = factory.Type().createReference("net.kyori.adventure.audience.Audience");
            } 
            else if (qName.equals("org.spongepowered.api.text.serializer.TextSerializers")) {
                // Logic: TextSerializers (static utility) moves to Adventure's serializer packages.
                // We map to LegacyComponentSerializer as a baseline, though specific field access 
                // (like JSON or FORMATTING_CODE) will need manual fixup or further method processors.
                newTypeRef = factory.Type().createReference("net.kyori.adventure.text.serializer.legacy.LegacyComponentSerializer");
            }

            if (newTypeRef != null) {
                // Preserve generics if any (though unlikely on these specific raw types)
                if (!candidate.getActualTypeArguments().isEmpty()) {
                    newTypeRef.setActualTypeArguments(candidate.getActualTypeArguments());
                }
                
                candidate.replace(newTypeRef);
                System.out.println("Refactored Type: " + qName + " -> " + newTypeRef.getQualifiedName() 
                    + " at line " + (candidate.getPosition().isValidPosition() ? candidate.getPosition().getLine() : "?"));
            }
        }
    }

    public static void main(String[] args) {
        // Default paths (editable by user)
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // CRITICAL SETTINGS for Source Preservation (Sniper Mode)
        // 1. Enable comments to prevent stripping
        launcher.getEnvironment().setCommentEnabled(true);
        
        // 2. Force Sniper Printer manually to preserve formatting
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );
        
        // 3. Defensive NoClasspath mode (Assume libraries are missing)
        launcher.getEnvironment().setNoClasspath(true);

        launcher.addProcessor(new SpongeTypeReplacementProcessor());

        try { 
            launcher.run(); 
            System.out.println("Refactoring complete. Check output in: " + outputPath);
        } catch (Exception e) { 
            e.printStackTrace(); 
        }
    }
}
