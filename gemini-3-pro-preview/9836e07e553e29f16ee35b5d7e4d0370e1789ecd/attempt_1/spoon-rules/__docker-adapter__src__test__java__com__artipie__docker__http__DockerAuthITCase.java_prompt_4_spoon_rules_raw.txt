import spoon.Launcher;

 *
 * This template strictly adheres to:
 * 1. Sniper Mode (Preserves formatting/comments)
 * 2. NoClasspath Compatibility (Defensive null checks)
 * 3. Java Generics Safety (Wildcards)

public class RefactoringTemplate {

    public static class ConfigurationProcessor extends AbstractProcessor<CtInvocation<?>> {

        // CONFIGURATION: Adjust these constants for the specific API change
        private static final String METHOD_NAME = "configure";
        private static final String TARGET_OWNER_PARTIAL = "Service"; // Partial match for NoClasspath safety
        private static final String NEW_WRAPPER_TYPE = "java.time.Duration";
        private static final String WRAPPER_METHOD = "ofMillis";

        @Override
        public boolean isToBeProcessed(CtInvocation<?> candidate) {
            // 1. Name Check
            // We check the method name first as it's the cheapest operation
            String name = candidate.getExecutable().getSimpleName();
            if (!METHOD_NAME.equals(name)) {
                return false;
            }

            // 2. Argument Count Check
            // We expect exactly 1 argument (int) that needs to be wrapped
            List<CtExpression<?>> args = candidate.getArguments();
            if (args.size() != 1) {
                return false;
            }

            // 3. Defensive Type Check (Critical for NoClasspath)
            // Spoon may return null for types if classpath is missing. 
            // We assume null types *might* be our target (best effort), 
            // but we skip if we are certain it is already the new type.
            CtExpression<?> arg = args.get(0);
            CtTypeReference<?> argType = arg.getType();

            if (argType != null) {
                // If it's already the new type, skip it to avoid double-wrapping
                if (argType.getQualifiedName().contains(NEW_WRAPPER_TYPE)) {
                    return false;
                }
                // If we know strictly it's not a primitive/number (e.g. it's a String), skip it
                if (!argType.isPrimitive() && !"java.lang.Integer".equals(argType.getQualifiedName())) {
                    // Logic allows int, Integer, or Unknown (null)
                    // You might want to skip here, but in NoClasspath, be careful filtering too aggressively.
                }
            }

            if (arg instanceof CtInvocation) {
                CtInvocation<?> argInv = (CtInvocation<?>) arg;
                if (WRAPPER_METHOD.equals(argInv.getExecutable().getSimpleName())) {
                    return false;
                }
            }

            // 4. Owner Check (Relaxed)
            // In NoClasspath, declaring type might be null or <unknown>.
            // We only filter OUT if we are sure it matches a different class.
            CtTypeReference<?> declaringType = candidate.getExecutable().getDeclaringType();
            if (declaringType != null && !declaringType.getQualifiedName().equals("<unknown>")) {
                if (!declaringType.getQualifiedName().contains(TARGET_OWNER_PARTIAL)) {
                    return false;
                }
            }

            return true;
        }

        @Override
        public void process(CtInvocation<?> invocation) {
            Factory factory = getFactory();
            CtExpression<?> originalArg = invocation.getArguments().get(0);

            // Create reference to the new wrapper type (e.g., java.time.Duration)
            CtTypeReference<?> wrapperRef = factory.Type().createReference(NEW_WRAPPER_TYPE);

                    wrapperRef,
                    factory.Type().voidPrimitiveType(), // return type of 'ofMillis' is actually Duration, but void is safe for ref creation
                    WRAPPER_METHOD,
                    factory.Type().integerPrimitiveType() // arg type
                ),
                originalArg.clone() // Clone the original argument to move it
            );

            // Replace the original argument with the wrapped version
            originalArg.replace(replacement);
            
            System.out.println("Refactored " + METHOD_NAME + " at line " + invocation.getPosition().getLine());
        }

        // Add the processor
        launcher.addProcessor(new ConfigurationProcessor());

        System.out.println("Starting Refactoring...");
        try {
            launcher.run();

        } catch (Exception e) {

    }
