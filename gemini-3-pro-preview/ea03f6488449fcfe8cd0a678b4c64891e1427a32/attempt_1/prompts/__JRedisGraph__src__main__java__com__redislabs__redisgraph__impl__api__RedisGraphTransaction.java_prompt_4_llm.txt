package org.example.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.code.CtInvocation;
import spoon.reflect.reference.CtTypeReference;
import spoon.support.sniper.SniperJavaPrettyPrinter;

/**
 * Refactoring script for migrating Redis Jedis library from version 3.x to 4.x/5.x.
 * 
 * Major Changes handled:
 * 1. Removal of `redis.clients.jedis.Client` class.
 *    - Strategy: Rename types to `redis.clients.jedis.Connection`.
 * 2. Removal/Renaming of accessors like `getClient()`.
 *    - Strategy: Rename `getClient()` invocations to `getConnection()`.
 * 3. Restructuring of `sendCommand` usage.
 *    - By migrating `Client` to `Connection`, chains like `jedis.getClient().sendCommand(...)` 
 *      become `jedis.getConnection().sendCommand(...)`, which is valid in the new API.
 */
public class JedisRefactoring {

    /**
     * Processor to handle the removal of the 'Client' class.
     * Replaces 'redis.clients.jedis.Client' with 'redis.clients.jedis.Connection'.
     */
    public static class ClientTypeProcessor extends AbstractProcessor<CtTypeReference<?>> {
        @Override
        public boolean isToBeProcessed(CtTypeReference<?> candidate) {
            // Defensive check for null
            if (candidate.getQualifiedName() == null) return false;
            
            // Check if it is the specific Redis Client class
            // Using contains/endsWith for robustness in NoClasspath environment
            return candidate.getQualifiedName().equals("redis.clients.jedis.Client");
        }

        @Override
        public void process(CtTypeReference<?> candidate) {
            // Change the name to Connection. 
            // We assume the package 'redis.clients.jedis' remains the same.
            candidate.setSimpleName("Connection");
            System.out.println("Refactored Type: Client -> Connection at line " + candidate.getPosition().getLine());
        }
    }

    /**
     * Processor to rename 'getClient()' calls to 'getConnection()'.
     * In Jedis 4, direct access to the helper Client is often replaced by Connection.
     */
    public static class GetClientMethodProcessor extends AbstractProcessor<CtInvocation<?>> {
        @Override
        public boolean isToBeProcessed(CtInvocation<?> candidate) {
            // Check method name
            return "getClient".equals(candidate.getExecutable().getSimpleName());
        }

        @Override
        public void process(CtInvocation<?> candidate) {
            // Rename the method invocation
            candidate.getExecutable().setSimpleName("getConnection");
            System.out.println("Refactored Method: getClient() -> getConnection() at line " + candidate.getPosition().getLine());
        }
    }

    public static void main(String[] args) {
        // Default paths (editable by user)
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        if (args.length > 0) inputPath = args[0];
        if (args.length > 1) outputPath = args[1];

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // CRITICAL SETTINGS for Robust Refactoring
        // 1. Enable comments to preserve manual notes
        launcher.getEnvironment().setCommentEnabled(true);
        
        // 2. Force Sniper Printer manually to preserve formatting of untouched code
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );
        
        // 3. Enable NoClasspath mode (robustness against missing dependencies)
        launcher.getEnvironment().setNoClasspath(true);

        // Add Processors
        launcher.addProcessor(new ClientTypeProcessor());
        launcher.addProcessor(new GetClientMethodProcessor());

        try {
            System.out.println("Starting Jedis Refactoring...");
            launcher.run();
            System.out.println("Refactoring complete. Output in: " + outputPath);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
