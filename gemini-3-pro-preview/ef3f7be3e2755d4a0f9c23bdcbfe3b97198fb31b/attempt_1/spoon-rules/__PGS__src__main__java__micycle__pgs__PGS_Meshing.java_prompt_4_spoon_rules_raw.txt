package org.tinspin.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.reference.CtTypeReference;
import spoon.support.sniper.SniperJavaPrettyPrinter;

public class TinSpinRefactoring {

    /**
     * Processor to handle the renaming of PointDistanceFunction to PointDistance.
     * This addresses the removal of PointDistanceFunction and the update of method signatures
     * in classes like CoverTree (e.g., create(int, double, PointDistanceFunction) -> create(int, double, PointDistance)).
     */
    public static class PointDistanceRenamer extends AbstractProcessor<CtTypeReference<?>> {

        @Override
        public boolean isToBeProcessed(CtTypeReference<?> candidate) {
            // Defensive check for null
            if (candidate == null || candidate.getSimpleName() == null) {
                return false;
            }

            // check for the specific class name
            // We use simple name check combined with a relaxed package check for NoClasspath safety
            if ("PointDistanceFunction".equals(candidate.getSimpleName())) {
                // If package info is present, verify it belongs to tinspin
                String qName = candidate.getQualifiedName();
                return qName.contains("org.tinspin") || qName.equals("PointDistanceFunction");
            }
            return false;
        }

        @Override
        public void process(CtTypeReference<?> candidate) {
            // Rename the type to PointDistance
            // This updates variable declarations, method arguments, return types, and casts.
            candidate.setSimpleName("PointDistance");
            
            // If the reference was fully qualified, ensure the package is correct.
            // PointDistanceFunction was in org.tinspin.index, PointDistance is also likely there 
            // based on the method signature: create(..., org.tinspin.index.PointDistance)
            if (candidate.getPackage() != null && candidate.getPackage().getQualifiedName().equals("org.tinspin.index")) {
                // The package is already correct, just the name changed.
            }
            
            System.out.println("Refactored PointDistanceFunction to PointDistance at " + 
                (candidate.getPosition().isValidPosition() ? candidate.getPosition().getLine() : "unknown line"));
        }
    }

    public static void main(String[] args) {
        // Default paths (can be overridden by args or hardcoded)
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // CRITICAL IMPLEMENTATION RULE 1: Preserve Source Code (Sniper Mode)
        // 1. Enable comments
        launcher.getEnvironment().setCommentEnabled(true);
        // 2. Force Sniper Printer manually
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );
        
        // CRITICAL IMPLEMENTATION RULE 2: NoClasspath Compatibility
        launcher.getEnvironment().setNoClasspath(true);

        // Add the processor
        launcher.addProcessor(new PointDistanceRenamer());

        try {
            launcher.run();
            System.out.println("Refactoring complete. Output in: " + outputPath);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
