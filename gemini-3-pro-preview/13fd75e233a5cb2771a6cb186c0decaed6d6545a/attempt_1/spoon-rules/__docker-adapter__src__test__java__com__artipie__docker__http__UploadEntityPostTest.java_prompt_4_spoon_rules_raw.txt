            CtExecutableReference<?> exec = candidate.getExecutable();
            if (exec == null) return false;

            // 2. Name Check: Target specific method name
            // REPLACE "deprecatedMethod" with the actual method name from your diff
            if (!"deprecatedMethod".equals(exec.getSimpleName())) {

            // if (candidate.getArguments().size() != 0) return false;

            // 4. Owner/Type Check (Defensive for NoClasspath)
            CtTypeReference<?> declaringType = exec.getDeclaringType();
            
            // In NoClasspath, declaringType might be null or <unknown>.
            // Use relaxed string matching instead of strict class resolution.
            if (declaringType != null) {
                String typeName = declaringType.getQualifiedName();
                // REPLACE "TargetClass" with the class name owning the method
                if (!typeName.contains("TargetClass") && !typeName.equals("<unknown>")) {
                    return false;
                }
            }

            return true;
        }

            // Example: Renaming the method
            CtExecutableReference<?> execRef = invocation.getExecutable();
            
            // Store old name for logging
            String oldName = execRef.getSimpleName();
            
            // Set new name
            // REPLACE "newMethod" with the new method name from your diff
            execRef.setSimpleName("newMethod");

        }
    }
