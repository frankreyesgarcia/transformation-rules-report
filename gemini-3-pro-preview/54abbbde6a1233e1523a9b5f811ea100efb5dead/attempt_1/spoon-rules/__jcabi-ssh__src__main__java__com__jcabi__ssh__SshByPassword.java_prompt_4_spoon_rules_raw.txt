package org.example.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.code.CtExpression;
import spoon.reflect.code.CtFieldRead;
import spoon.reflect.code.CtTypeAccess;
import spoon.reflect.reference.CtTypeReference;
import spoon.support.sniper.SniperJavaPrettyPrinter;

import java.util.HashMap;
import java.util.Map;

public class TvRefactoring {

    /**
     * Processor to handle the removal of com.jcabi.aspects.Tv.
     * Replaces field accesses (e.g., Tv.SEVEN) with their literal values.
     */
    public static class TvProcessor extends AbstractProcessor<CtFieldRead<?>> {
        
        private static final Map<String, Integer> KNOWN_CONSTANTS = new HashMap<>();
        
        static {
            // Based on the specific diff provided: Tv.SEVEN -> 7
            KNOWN_CONSTANTS.put("SEVEN", 7);
            
            // Since the entire class 'Tv' (Test Values) is removed, usually implies standard constants 
            // are also gone. We include common mappings to make the refactoring robust for the whole class removal.
            KNOWN_CONSTANTS.put("ZERO", 0);
            KNOWN_CONSTANTS.put("ONE", 1);
            KNOWN_CONSTANTS.put("TWO", 2);
            KNOWN_CONSTANTS.put("THREE", 3);
            KNOWN_CONSTANTS.put("FOUR", 4);
            KNOWN_CONSTANTS.put("FIVE", 5);
            KNOWN_CONSTANTS.put("SIX", 6);
            KNOWN_CONSTANTS.put("EIGHT", 8);
            KNOWN_CONSTANTS.put("NINE", 9);
            KNOWN_CONSTANTS.put("TEN", 10);
            // Tv.MILLION, etc., could be added here if known, but we stick to the diff context + reasonable bounds.
        }

        @Override
        public boolean isToBeProcessed(CtFieldRead<?> candidate) {
            // 1. Target Check (Defensive for NoClasspath)
            CtExpression<?> target = candidate.getTarget();
            if (!(target instanceof CtTypeAccess)) {
                return false;
            }

            // 2. Class Name Check
            // We look for "Tv" or "com.jcabi.aspects.Tv"
            CtTypeReference<?> accessedType = ((CtTypeAccess<?>) target).getAccessedType();
            if (accessedType == null) return false;
            
            String qualifiedName = accessedType.getQualifiedName();
            if (qualifiedName == null || 
               (!qualifiedName.contains("com.jcabi.aspects.Tv") && !qualifiedName.equals("Tv"))) {
                return false;
            }

            // 3. Field Name Check
            // Check if the field being accessed is one of our known constants (specifically SEVEN)
            String fieldName = candidate.getVariable().getSimpleName();
            return KNOWN_CONSTANTS.containsKey(fieldName);
        }

        @Override
        public void process(CtFieldRead<?> fieldRead) {
            String fieldName = fieldRead.getVariable().getSimpleName();
            Integer value = KNOWN_CONSTANTS.get(fieldName);

            if (value != null) {
                // Create a literal (e.g., 7)
                // We use integer primitive type
                CtExpression<?> replacement = getFactory().Code().createLiteral(value);
                
                // preserve comments from the original field read if any
                replacement.setComments(fieldRead.getComments());
                
                // Replace Tv.SEVEN with 7
                fieldRead.replace(replacement);
                
                System.out.println("Refactored Tv." + fieldName + " to literal " + value + 
                                   " at line " + fieldRead.getPosition().getLine());
            }
        }
    }

    public static void main(String[] args) {
        // Default paths (can be overridden or hardcoded)
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // CRITICAL SETTINGS FOR SPOON
        
        // 1. Enable comments to preserve them during transformation
        launcher.getEnvironment().setCommentEnabled(true);
        
        // 2. Force Sniper Printer manually to preserve formatting of untouched code
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );
        
        // 3. Enable NoClasspath mode (robustness against missing dependencies)
        launcher.getEnvironment().setNoClasspath(true);

        // Add the processor
        launcher.addProcessor(new TvProcessor());

        // Run transformation
        try {
            System.out.println("Starting Tv.SEVEN refactoring...");
            launcher.run();
            System.out.println("Refactoring complete.");
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
