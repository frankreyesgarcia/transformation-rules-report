package org.example.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.declaration.CtImport;
import spoon.reflect.reference.CtTypeReference;
import spoon.support.sniper.SniperJavaPrettyPrinter;

public class PubSubLiteRefactoring {

    /**
     * Processor to handle the removal of com.google.cloud.pubsublite.PublishMetadata.
     * The library replaced usage of PublishMetadata in the Publisher API (ApiFuture<PublishMetadata>)
     * with a simple String message ID (ApiFuture<String>) or similar String-based representations.
     *
     * This processor locates references to PublishMetadata and replaces them with java.lang.String.
     */
    public static class PublishMetadataProcessor extends AbstractProcessor<CtTypeReference<?>> {

        @Override
        public boolean isToBeProcessed(CtTypeReference<?> candidate) {
            // 1. Defensive Checks
            if (candidate == null || candidate.getQualifiedName() == null) {
                return false;
            }

            // 2. Identify the target removed class
            // We use string matching to be safe in NoClasspath environments
            if (!candidate.getQualifiedName().equals("com.google.cloud.pubsublite.PublishMetadata")) {
                return false;
            }

            // 3. Context Check
            // Do not replace the reference if it is part of an import statement.
            // We want to update variable declarations, return types, and generic arguments,
            // but rewriting "import com...PublishMetadata" to "import java.lang.String" is invalid/redundant.
            if (candidate.isParentInitialized() && candidate.getParent() instanceof CtImport) {
                return false;
            }

            return true;
        }

        @Override
        public void process(CtTypeReference<?> candidate) {
            // Transformation: Replace 'PublishMetadata' with 'String'
            // This aligns with the API change where publish futures now return the Message ID (String).
            CtTypeReference<?> stringRef = getFactory().Type().createReference("java.lang.String");
            
            // Replacing the element in the AST
            candidate.replace(stringRef);
            
            System.out.println("Refactored PublishMetadata to String at line " 
                + (candidate.getPosition().isValidPosition() ? candidate.getPosition().getLine() : "unknown"));
        }
    }

    public static void main(String[] args) {
        // Default paths (editable by user)
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // CRITICAL SETTINGS for Spoon 11+ and robust refactoring
        
        // 1. Enable comments to preserve license headers and Javadocs
        launcher.getEnvironment().setCommentEnabled(true);
        
        // 2. Force Sniper Printer manually to preserve formatting of untouched code
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );
        
        // 3. Configure NoClasspath mode (assumes dependencies are missing)
        launcher.getEnvironment().setNoClasspath(true);

        // Add the processor
        launcher.addProcessor(new PublishMetadataProcessor());

        try {
            System.out.println("Starting refactoring...");
            launcher.run();
            System.out.println("Refactoring complete. Check output at: " + outputPath);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
