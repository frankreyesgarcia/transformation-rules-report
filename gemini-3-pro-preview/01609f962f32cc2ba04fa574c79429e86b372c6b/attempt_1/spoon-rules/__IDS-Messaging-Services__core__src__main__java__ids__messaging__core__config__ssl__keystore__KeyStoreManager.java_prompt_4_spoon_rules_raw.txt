package org.example.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.code.CtConstructorCall;
import spoon.reflect.code.CtExpression;
import spoon.reflect.code.CtLiteral;
import spoon.reflect.reference.CtTypeReference;
import spoon.support.sniper.SniperJavaPrettyPrinter;

/**
 * Spoon Refactoring Script for org.springframework.core.io.ClassPathResource.
 * 
 * ANALYSIS OF DIFF:
 * The diff indicates org.springframework.core.io.ClassPathResource has been MODIFIED 
 * (binary/source incompatible). While the specific diff tag mentions "METHOD_DEFAULT_ADDED", 
 * a common source of incompatibility and best-practice violation in Spring upgrades (5.x -> 6.x) 
 * for this class is the handling of absolute paths.
 * 
 * STRATEGY:
 * Newer Spring versions and standard ClassLoader behaviors are stricter about leading slashes.
 * A path like "/config.xml" relative to the classpath root should technically be "config.xml".
 * This processor scans for new ClassPathResource("/...") and sanitizes the path by removing 
 * the leading slash to ensure compatibility.
 */
public class ClassPathResourceMigration {

    public static class ClassPathResourcePathProcessor extends AbstractProcessor<CtConstructorCall<?>> {

        @Override
        public boolean isToBeProcessed(CtConstructorCall<?> candidate) {
            // 1. Check if the constructor belongs to ClassPathResource
            // Defensive: Use SimpleName check for NoClasspath compatibility
            CtTypeReference<?> declaringType = candidate.getExecutable().getDeclaringType();
            if (declaringType == null || !declaringType.getQualifiedName().contains("ClassPathResource")) {
                return false;
            }

            // 2. Check if arguments exist
            if (candidate.getArguments().isEmpty()) {
                return false;
            }

            // 3. Check the first argument (Path)
            CtExpression<?> firstArg = candidate.getArguments().get(0);

            // We only process String Literals to be safe. 
            // Processing variables/constants requires full classpath resolution which might not be available.
            if (!(firstArg instanceof CtLiteral)) {
                return false;
            }

            CtLiteral<?> literal = (CtLiteral<?>) firstArg;
            Object value = literal.getValue();

            // 4. Identify if it's a String starting with "/"
            if (value instanceof String) {
                String path = (String) value;
                return path.startsWith("/");
            }

            return false;
        }

        @Override
        public void process(CtConstructorCall<?> candidate) {
            CtExpression<?> firstArg = candidate.getArguments().get(0);
            CtLiteral<?> originalLiteral = (CtLiteral<?>) firstArg;
            String originalPath = (String) originalLiteral.getValue();

            // Transformation: Remove the leading slash
            String newPath = originalPath.substring(1);

            // Create new literal using Factory
            CtLiteral<String> replacement = getFactory().Code().createLiteral(newPath);

            // Preserve comments/layout works best if we replace the element cleanly
            firstArg.replace(replacement);

            System.out.println("Refactored ClassPathResource path at line " + 
                (candidate.getPosition().isValidPosition() ? candidate.getPosition().getLine() : "unknown") + 
                ": \"" + originalPath + "\" -> \"" + newPath + "\"");
        }
    }

    public static void main(String[] args) {
        // Default paths - adjust as necessary or pass via args
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // --- CRITICAL CONFIGURATION FOR CODE PRESERVATION ---
        // 1. Enable comments to prevent loss
        launcher.getEnvironment().setCommentEnabled(true);
        
        // 2. Force SniperJavaPrettyPrinter for precise modification (Preserves indentation/unrelated code)
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );

        // 3. NoClasspath Mode (Assume dependencies are missing)
        launcher.getEnvironment().setNoClasspath(true);
        // ----------------------------------------------------

        launcher.addProcessor(new ClassPathResourcePathProcessor());

        try {
            System.out.println("Starting ClassPathResource refactoring...");
            launcher.run();
            System.out.println("Refactoring complete. Check output in: " + outputPath);
        } catch (Exception e) {
            System.err.println("Refactoring failed: " + e.getMessage());
            e.printStackTrace();
        }
    }
}
