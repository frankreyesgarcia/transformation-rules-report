package org.example.migration;

import spoon.Launcher;
import spoon.processing.AbstractProcessor;
import spoon.reflect.code.CtConstructorCall;
import spoon.reflect.code.CtExpression;
import spoon.reflect.code.CtComment;
import spoon.reflect.reference.CtTypeReference;
import spoon.support.sniper.SniperJavaPrettyPrinter;

public class DirectoryScannerRefactoring {

    /**
     * Processor to identify and annotate usages of the DirectoryScanner constructor.
     * Based on the diff, this element is being tracked (marked UNCHANGED in report, 
     * but potentially flagged for review in the context of dependency analysis).
     * 
     * Strategy: Locate usages of new DirectoryScanner(File, TestListResolver) 
     * and add a 'Review' comment to ensure compatibility with the plugin version.
     */
    public static class DirectoryScannerProcessor extends AbstractProcessor<CtConstructorCall<?>> {

        @Override
        public boolean isToBeProcessed(CtConstructorCall<?> candidate) {
            // 1. Check Owner/Type Name
            // In NoClasspath, we rely on the reference name or qualified name string.
            CtTypeReference<?> declaringType = candidate.getExecutable().getDeclaringType();
            if (declaringType == null) return false;
            
            String qualifiedName = declaringType.getQualifiedName();
            if (!qualifiedName.contains("DirectoryScanner")) {
                return false;
            }
            
            // Check package specifically if available to distinguish from other DirectoryScanners
            if (!qualifiedName.contains("org.apache.maven.plugin.surefire.util") && !qualifiedName.equals("DirectoryScanner")) {
                return false;
            }

            // 2. Argument Count Check (2 arguments based on diff)
            if (candidate.getArguments().size() != 2) return false;

            // 3. Argument Type Checks (Defensive for NoClasspath)
            
            // Arg 0: java.io.File
            CtExpression<?> arg0 = candidate.getArguments().get(0);
            CtTypeReference<?> type0 = arg0.getType();
            // If type is known and DOES NOT contain File, skip. If null (unknown), proceed cautiously.
            if (type0 != null && !type0.getQualifiedName().contains("File")) {
                return false;
            }

            // Arg 1: org.apache.maven.surefire.api.testset.TestListResolver
            CtExpression<?> arg1 = candidate.getArguments().get(1);
            CtTypeReference<?> type1 = arg1.getType();
            // If type is known and DOES NOT contain TestListResolver, skip.
            if (type1 != null && !type1.getQualifiedName().contains("TestListResolver")) {
                return false;
            }

            return true;
        }

        @Override
        public void process(CtConstructorCall<?> candidate) {
            // Since the diff states UNCHANGED, we do not rename or delete.
            // However, to demonstrate the refactoring capability requested by the prompt,
            // we will inject a comment to flag this constructor usage for review.
            
            getFactory().getEnvironment().setAutoImports(true);
            
            candidate.addComment(
                getFactory().Code().createComment(
                    "REVIEW: Ensure compatibility with org.apache.maven.plugin.surefire.util.DirectoryScanner", 
                    CtComment.CommentType.BLOCK
                )
            );
            
            System.out.println("Flagged DirectoryScanner usage at " + candidate.getPosition());
        }
    }

    public static void main(String[] args) {
        // Default paths (editable by user)
        String inputPath = "./src/main/java";
        String outputPath = "./src/main/gen";

        Launcher launcher = new Launcher();
        launcher.addInputResource(inputPath);
        launcher.setSourceOutputDirectory(outputPath);

        // ===========================================
        // CRITICAL IMPLEMENTATION RULES (Sniper Mode)
        // ===========================================
        
        // 1. Enable comments to ensure they are preserved and new ones are printed
        launcher.getEnvironment().setCommentEnabled(true);
        
        // 2. Force Sniper Printer manually to preserve exact formatting/indentation
        launcher.getEnvironment().setPrettyPrinterCreator(
            () -> new SniperJavaPrettyPrinter(launcher.getEnvironment())
        );
        
        // 3. Set NoClasspath to true (Defensive mode)
        launcher.getEnvironment().setNoClasspath(true);

        launcher.addProcessor(new DirectoryScannerProcessor());
        
        try {
            launcher.run();
            System.out.println("Refactoring complete.");
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
